### 查询"手机"

```java
GET /product_index/_search
{
  "query": {
    "match": {
      "name": "手机"
    }
  },
  //分页
  "from": 2,
  "size": 2,
  //排序
  "sort": [
    {
      "price": {
        "order": "asc"
      }
    }
  ],
  //指定显示多个字段 : ["name","price"]
  "_source": "name",
  //过滤
  "filter": {
  	"range": {
			"price": {
      	"gte": 10,
        "lte": 50
      }
    }
  },
  //高亮显示
  "highlight": {
    "fields": {
      "name": {
        "pre_tags": "<em>",
        "post_tags": "</em>"
      }
    }
  }
}
```

### bool

```java
//类似and
GET /product_index/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "match": {
            "name": "手机"
          }
        },
        {
          "match": {
            "price": "20"
          }
        }
      ]
    }
  }
}

//类似or
GET /product_index/_search
{
  "query": {
    "bool": {
      "should": [
        {
          "match": {
            "name": "手机"
          }
        },
        {
          "match": {
            "price": "20"
          }
        }
      ]
    }
  }
}

//类似not
GET /product_index/_search
{
  "query": {
    "bool": {
      "must_not": [
        {
          "match": {
            "name": "手机"
          }
        }
      ]
    }
  }
}
```

### 不分词查询

#### term

```java
GET /product_index/_search
{
  "query": {
    //直接查找"手机"这个词的倒排搜索,不会对"手机"这个词进行分词查询
    //默认情况分词器是'Standard Analyzer',这样就会导致即使name等于"手机",也会查不出来
    "term": {
      "name": {
        "value": "手机"
      }
    }
  }
}
```

#### keyword

```java
GET /product_index/_search
{
  "query": {
    "match": {
      //如果name字段是keyword类型,那么这里不会进行分词查询
      "name": "手机"
    }
  }
}
```
