### 高亮显示

```java
GET /product_index/_search
{
  "query": {
    "match": {
      "name": "手机"
    }
  },
  "highlight": {
    "fields": {
      "name": {
        "pre_tags": "<em>",
        "post_tags": "</em>"
      }
    }
  }
}
```

### keyword类型

```java
//不会进行分词查询
```

### match

```java
GET /product_index/_search
{
  "query": {
    "match": {
      "name": "手机"
    }
  }
}
```

### 只显示指定字段

```java
GET /product_index/_search
{
  "query": {
    "match": {
      "name": "手机"
    }
  },
  //显示多列 : ["name","price"]
  "_source": "name"
}
```

### term

```java
//直接查找"手机"这个词的倒排搜索,不会将"手机"进行分词查询
GET /user_index/_search
{
  "query": {
    "term": {
      "name": {
        "value": "手机"
      }
    }
  }
}
```

### 过滤
```java
GET /product_index/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "match": {
            "name": "手机"
          }
        }
      ],
      "filter": {
        "range": {
          "age": {
            "gte": 10,
            "lte": 50
          }
        }
      }
    }
  }
}
```

### bool

```java
//and查询
GET /product_index/_search
{
  "query": {
    "bool": {
      "must": [
        {
          "match": {
            "name": "手机"
          }
        },
        {
          "match": {
            "price": "20"
          }
        }
      ]
    }
  }
}

//or查询
GET /product_index/_search
{
  "query": {
    "bool": {
      "should": [
        {
          "match": {
            "name": "手机"
          }
        },
        {
          "match": {
            "price": "20"
          }
        }
      ]
    }
  }
}

//not查询
GET /product_index/_search
{
  "query": {
    "bool": {
      "must_not": [
        {
          "match": {
            "name": "手机"
          }
        }
      ]
    }
  }
}
```

### 分页

```java
GET /product_index/_search
{
  "query": {
    "match": {
      "name": "手机"
    }
  },
```

### 排序

```java

      "name": "手机"
    }
  },
  "sort": [
```
